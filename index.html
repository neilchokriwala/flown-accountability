    <script>

        let currentDate = new Date();

        let selectedDate = null;

        let links = {};


        // Load saved links from localStorage

        function loadLinks() {

            const saved = localStorage.getItem('accountabilityLinks');

            if (saved) {

                links = JSON.parse(saved);

            }

        }


        // Save links to localStorage

        function saveLinks() {

            localStorage.setItem('accountabilityLinks', JSON.stringify(links));

        }


        // Format date as key for storage

        function dateKey(date) {

            return `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}-${String(date.getDate()).padStart(2, '0')}`;

        }


        // Generate wall name for a date

        function generateWallName(date) {

            const year = date.getFullYear();

            const month = String(date.getMonth() + 1).padStart(2, '0');

            const day = String(date.getDate()).padStart(2, '0');

            return `flown-${year}-${month}-${day}`;

        }


        // Render the calendar

        function renderCalendar() {

            const year = currentDate.getFullYear();

            const month = currentDate.getMonth();

            

            // Update month/year display

            const monthNames = ['January', 'February', 'March', 'April', 'May', 'June',

                              'July', 'August', 'September', 'October', 'November', 'December'];

            document.getElementById('monthYear').textContent = `${monthNames[month]} ${year}`;

            

            // Clear days

            const daysContainer = document.getElementById('days');

            daysContainer.innerHTML = '';

            

            // Get first day of month and number of days

            const firstDay = new Date(year, month, 1).getDay();

            const daysInMonth = new Date(year, month + 1, 0).getDate();

            const today = new Date();

            

            // Add empty cells for days before month starts

            for (let i = 0; i < firstDay; i++) {

                const emptyDay = document.createElement('div');

                emptyDay.className = 'day other-month';

                daysContainer.appendChild(emptyDay);

            }

            

            // Add days of the month

            for (let day = 1; day <= daysInMonth; day++) {

                const dayElement = document.createElement('div');

                dayElement.className = 'day';

                

                const date = new Date(year, month, day);

                const key = dateKey(date);

                

                // Check if this is today

                if (date.toDateString() === today.toDateString()) {

                    dayElement.classList.add('today');

                }

                

                // Check if this day has a link

                if (links[key]) {

                    dayElement.classList.add('has-link');

                    const indicator = document.createElement('div');

                    indicator.className = 'day-indicator';

                    dayElement.appendChild(indicator);

                }

                

                const dayNumber = document.createElement('div');

                dayNumber.className = 'day-number';

                dayNumber.textContent = day;

                dayElement.appendChild(dayNumber);

                

                dayElement.onclick = () => openModal(date);

                

                daysContainer.appendChild(dayElement);

            }

        }


        // Navigation functions

        function previousMonth() {

            currentDate.setMonth(currentDate.getMonth() - 1);

            renderCalendar();

        }


        function nextMonth() {

            currentDate.setMonth(currentDate.getMonth() + 1);

            renderCalendar();

        }


        function currentMonth() {

            currentDate = new Date();

            renderCalendar();

        }


        // Modal functions

        function openModal(date) {

            selectedDate = date;

            const key = dateKey(date);

            const modal = document.getElementById('dayModal');

            const modalDate = document.getElementById('modalDate');

            const existingSection = document.getElementById('existingLinkSection');

            const newSection = document.getElementById('newWallSection');

            const saveBtn = document.getElementById('saveBtn');

            

            // Format the date for display

            const options = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };

            modalDate.textContent = date.toLocaleDateString('en-US', options);

            

            // Check if link exists

            if (links[key]) {

                existingSection.style.display = 'block';

                newSection.style.display = 'none';

                saveBtn.style.display = 'none';

                document.getElementById('existingLink').href = links[key];

                document.getElementById('existingLink').textContent = links[key];

            } else {

                // Show new wall section

                existingSection.style.display = 'none';

                newSection.style.display = 'block';

                saveBtn.style.display = 'block';

                

                const wallName = generateWallName(date);

                document.getElementById('wallNameDisplay').textContent = wallName;

                document.getElementById('linkInput').value = '';

                

                // Copy to clipboard and open AccountabilityWall.com

                navigator.clipboard.writeText(wallName).then(() => {

                    // Small delay to ensure user sees the modal before new tab opens

                    setTimeout(() => {

                        window.open('https://accountabilitywall.com', '_blank');

                    }, 500);

                });

            }

            

            modal.classList.add('active');

        }


        function closeModal() {

            document.getElementById('dayModal').classList.remove('active');

        }


        function saveLink() {

            const link = document.getElementById('linkInput').value.trim();

            

            if (link) {

                const key = dateKey(selectedDate);

                links[key] = link;

                saveLinks();

                renderCalendar();

                closeModal();

            }

        }


        function removeLink() {

            if (confirm('Remove the accountability wall link for this day?')) {

                const key = dateKey(selectedDate);

                delete links[key];

                saveLinks();

                renderCalendar();

                closeModal();

            }

        }


        function openExistingLink() {

            const link = document.getElementById('existingLink').href;

            window.open(link, '_blank');

        }


        // Close modal when clicking outside

        window.onclick = function(event) {

            const modal = document.getElementById('dayModal');

            if (event.target === modal) {

                closeModal();

            }

        }


        // Initialize

        loadLinks();

        renderCalendar();

    </script>
